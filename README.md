try begins ThreeJS with TypeScript
<h1>How to Run</h1>
<ul>
    <li>
        npm install -g typescript
    </li>
    <li>
        npm install
    </li>
    <li>
        tsc -p ./src/server
    </li>
    <li>
        tsc -p ./src/client
    </li>
    <li>
        node ./dist/server/server.js
    </li>
</ul>
